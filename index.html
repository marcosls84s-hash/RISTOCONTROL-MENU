<body style="text-align:center; padding:50px; font-family:sans-serif;">

    <h1 style="color:#FF9800;">RISTOCONTROL MENU</h1>

    <p id="stato">Connessione al database...</p>

    <div id="lista" style="text-align:left; max-width:400px; margin:auto;"></div>


    <script src="https://www.gstatic.com/firebasejs/8.10.1/firebase-app.js"></script>

    <script src="https://www.gstatic.com/firebasejs/8.10.1/firebase-database.js"></script>

    <script>

        const config = { databaseURL: "https://comandaristorante-facile-default-rtdb.europe-west1.firebasedatabase.app" };

        firebase.initializeApp(config);

        const resId = new URLSearchParams(window.location.search).get('res');


        if (!resId) {

            document.getElementById('stato').innerText = "ERRORE: ID MANCANTE";

        } else {

            firebase.database().ref("restaurants/" + resId).on("value", (snap) => {

                const data = snap.val();

                if (data && data.menu) {

                    document.getElementById('stato').style.display = "none";

                    let html = "";

                    Object.values(data.menu).forEach(p => {

                        html += `<p><b>${p.name}</b>: €${p.price.toFixed(2)}</p>`;

                    });

                    document.getElementById('lista').innerHTML = html;

                } else {

                    document.getElementById('stato').innerText = "MENU NON TROVATO PER ID: " + resId;

                }

            });

        }

    </script>

</body>

</html>
